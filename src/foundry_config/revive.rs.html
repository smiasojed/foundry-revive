15#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize, Default)]
16/// Revive Config
17pub struct ReviveConfig {
18    /// Enable compilation using revive
19    pub revive_compile: bool,
20    /// The revive path
21    pub revive_path: Option<PathBuf>,
22    /// solc path to use along the revive compiler
23    pub solc_path: Option<PathBuf>,

25
26impl ReviveConfig {
27    /// Returns the `ProjectPathsConfig` sub set of the config.
28    pub fn project_paths(config: &Config) -> ProjectPathsConfig<MultiCompilerLanguage> {
29        let builder = ProjectPathsConfig::builder()
30            .cache(config.cache_path.join(REVIVE_SOLIDITY_FILES_CACHE_FILENAME))
31            .sources(&config.src)
32            .tests(&config.test)
33            .scripts(&config.script)
34            .artifacts(config.root.join(REVIVE_ARTIFACTS_DIR))
35            .libs(config.libs.iter())
36            .remappings(config.get_all_remappings())
37            .allowed_path(&config.root)
38            .allowed_paths(&config.libs)
39            .allowed_paths(&config.allow_paths)
40            .include_paths(&config.include_paths);
41
42        builder.build_with_root(&config.root)
43    }
